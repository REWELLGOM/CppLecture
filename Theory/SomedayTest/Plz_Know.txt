namespace Swap
{   
   void swapFunc(char *ch1, char *ch2)
    {
        cout<<*ch2<<endl; //ch2만으로 출력하면 ch1의 값을 포함하여 연속된 메모리 영역을 출력
        char ch = *ch1;
        *ch1 = *ch2;
        *ch2 = ch;
    }
}
int main(void)
{
    char ch1 = 'a';
    char ch2 = 'b';
    Swap::swapFunc(&ch1, &ch2);
    cout<<ch1<<' '<<ch2<<endl;
    return 0;
}
